如图，实现box相对于container的上下左右居中。

    
    .container{
     position: relative;
     margin: 0 auto;
     width: 300px;
     height: 200px;
     border: 1px solid red;
     }
    
     .box{
     position: absolute;
     left: 50%;
     top: 50%;
     margin-left: -50px;
     margin-top: -50px;
     width: 100px;
     height: 100px;
     border: 1px solid blue;
     }
